const SHEET_NAME = 'orders'; function doGet(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const data = sheet.getDataRange().getValues(); const rows = data.slice(1).map(r => ({ num: r[0], items: JSON.parse(r[1] || '[]'), done: r[2], time: r[3], payType: r[4] || '', total: r[5] || 0 })); return ContentService.createTextOutput(JSON.stringify(rows)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME); const payload = JSON.parse(e.postData.contents); if (payload.action === 'add') { sheet.appendRow([payload.num, JSON.stringify(payload.items), false, payload.time, payload.payType || '', payload.total || 0]); } else if (payload.action === 'update') { const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == payload.num) { sheet.getRange(i + 1, 3).setValue(payload.done); break; } } } else if (payload.action === 'reset') { const lastRow = sheet.getLastRow(); if (lastRow > 1) sheet.deleteRows(2, lastRow - 1); } return ContentService.createTextOutput(JSON.stringify({ok: true})) .setMimeType(ContentService.MimeType.JSON); }
